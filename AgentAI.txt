        const savedBase = localStorage.getItem("conv_api_base");
        const savedToken = localStorage.getItem("conv_api_token");
        const savedUserId = localStorage.getItem("conv_api_user_id");
        const defaultBase = savedBase || window.location.origin;
        apiBaseInput.value = defaultBase;
        tokenInput.value = savedToken || "";
        userIdInput.value = savedUserId || "";
        agentIdInput.value = (hiddenId && hiddenId.value) ? hiddenId.value : "";

        function appendMessage(type, text) {
            const div = document.createElement("div");
            div.className = "conversation-msg " + type;
            div.textContent = text;
            messageBox.appendChild(div);
            messageBox.scrollTop = messageBox.scrollHeight;
        }

        function openPopup() {
            popup.classList.add("open");
            popup.setAttribute("aria-hidden", "false");
            input.focus();
        }

        function closePopup() {
            popup.classList.remove("open");
            popup.setAttribute("aria-hidden", "true");
        }

        launcher.addEventListener("click", function () {
            if (popup.classList.contains("open")) {
                closePopup();
            } else {
                openPopup();
            }
        });

        closeBtn.addEventListener("click", closePopup);

        async function sendQuery() {
            const text = (input.value || "").trim();
            const base = (apiBaseInput.value || "").trim().replace(/\/+$/, "");
            const agentId = (agentIdInput.value || "").trim();
            const token = (tokenInput.value || "").trim();
            const userId = (userIdInput.value || "").trim();

            if (!text) {
                return;
            }
            if (!base) {
                appendMessage("error", "API Base URL is required.");
                return;
            }
            if (!agentId) {
                appendMessage("error", "Agent ID is required.");
                return;
            }
            if (!token) {
                appendMessage("error", "Bearer Token is required.");
                return;
            }

            localStorage.setItem("conv_api_base", base);
            localStorage.setItem("conv_api_token", token);
            localStorage.setItem("conv_api_user_id", userId);

            appendMessage("user", text);
            input.value = "";
            sendBtn.disabled = true;

            const headers = {
                "Content-Type": "application/json",
                "Authorization": "Bearer " + token
            };
            if (userId) {
                headers["X-User-Id"] = userId;
            }

            try {
                const response = await fetch(base + "/query", {
                    method: "POST",
                    headers: headers,
                    body: JSON.stringify({
                        id: agentId,
                        channel_name: "TEXT",
                        channel_type: "MARKETING",
                        query: text
                    })
                });

                const data = await response.json().catch(function () { return {}; });
                if (!response.ok) {
                    const err = (data && data.detail) ? data.detail : ("HTTP " + response.status);
                    appendMessage("error", String(err));
                    return;
                }

                appendMessage("agent", data.answer || "(No answer returned)");
            } catch (err) {
                appendMessage("error", "Request failed: " + (err && err.message ? err.message : "Unknown error"));
            } finally {
                sendBtn.disabled = false;
                input.focus();
            }
        }

        sendBtn.addEventListener("click", sendQuery);
        input.addEventListener("keydown", function (e) {
            if (e.key === "Enter" && !e.shiftKey) {
                e.preventDefault();
                sendQuery();
            }
        });
    })();
</script>
